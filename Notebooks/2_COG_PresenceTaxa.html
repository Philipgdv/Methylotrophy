Taxonomy = pd.DataFrame(index=Root_COGsOI_Counts.columns)
Taxonomy["Kingdom"] = "NaN"
Taxonomy["Order"] = "NaN"

for ID in Taxonomy.index:

        Taxonomy.Order[ID] = ncbi.get_lineage(ID)[3]

        Taxonomy.Order[ID] = 1935183
